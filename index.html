<input type="text" placeholder="IPA tones typer" id="tones" autocomplete="off"/>
<button class="smallbutton" onclick="enter('\u02E9')">1</button>
<button class="smallbutton" onclick="enter('\u02E8')">2</button>
<button class="smallbutton" onclick="enter('\u02E7')">3</button>
<button class="smallbutton" onclick="enter('\u02E6')">4</button>
<button class="smallbutton" onclick="enter('\u02E5')">5</button>
<button onclick="document.getElementById('tones').value = ''" style="color:red">X</button>
<button onclick="copytext()" style="color:blue">C</button>
<script>
function enter(character){
document.getElementById("tones").value+=character;
document.getElementById("tones").focus();
}
function copytext() {
  var copyText = document.getElementById("tones");
  copyText.select();
  copyText.setSelectionRange(0, 99999)
  document.execCommand("copy");
}
</script>
